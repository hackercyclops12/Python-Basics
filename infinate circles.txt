from random import*

from tkinter import*

size = 500

window = Tk()

canvas = Canvas(window, width=size, height=size)
 #replace width with x variable 1920 and height with y of 1080 for full screen circles! 
canvas.pack()

while True:
	
	col = choice(['pink','orange','purple','yellow'])
	
	x0 = randint(0, size)
 # replace  size var with x var for full screen circles	
	y0 = randint(0, size)
 # replace  size var with y var for full screen circles	
	d = randint(0, size/5)
	
	canvas.create_oval(x0, y0, x0 + d, y0 + d, fill=col)

	window.update()
