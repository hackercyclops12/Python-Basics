import turtle 
import math
import colorsys
phi = 180*(3-math.sqrt(5))
t = turtle.pen()
turtle.speed(0)
def square (t, size):
    for tmp in range(0,4):
	t.forward(size)
	t.right(90)
num = 200

for x in reversed(range(0, num)):
    turtle.fillcolor(colorsys.hsv_to_rgb(x/num, 1.0, 1.0))
    turtle.begin_fill()
    turtle.circle(5 + x, None, 11)
    square(t, 5 + x)
    turtle.end_fill()
    turtle.right(phi)
    turtle.right(.8)
